GAUSS JACOBI

Iterative Solutions for Linear Systems

Convergence Requirements

1. Strictly Diagonally Dominant

Absolute value of diagonal elements is greater than the sum of other row elements.

2. Symmetric Positive Definite

System converges if the matrix is symmetric and all eigenvalues are positive.

3. Spectral Radius Condition

Iterative stability is guaranteed when the spectral radius is less than one.

Matrix Iteration Form

x(k+1) = D-1 (b-(L+U)x(k))

Jacobi Iteration Matrix

Tj=-D-1(L+U)

x0 = np.zeros(3)
x0 = list(map(float, input("Enter x0 x1 x2: ").split()))
tol = float(input("Enter the required tolerance : "))
n = A.shape[0]

# Decomposition logic
D = np.diag(np.diag(A))
L = np.tril(A, -1)
U = np.triu(A, 1)

# Jacobi iteration matrix analysis
Tj = -np.linalg.inv(D) @ (L+U)
rho = max(abs(np.linalg.eigvals(Tj)))
print("Spectral radius =", rho)

if rho < 1:
    print("Jacobi method will converge")
    solution = gauss_jacobi(A, b, x0, tol)
    print("Solution:", solution)
else:
    print("Jacobi method will not converge")

624112: The Solver Logic

def gauss_jacobi(A, b, x0, tol):
    n = len(b)
    x = np.array(x0).copy()
    x_new = np.zeros(n)
    print("Initial guess: ", x)
    x_list.append(float(x[0]))
    y_list.append(float(x[1]))
    z_list.append(float(x[2]))
    k = 0; k_list.append(int(k))
    while True:
        print(f"\n Iteration {k+1}")
        k += 1; k_list.append(int(k))
        for i in range(n):
            s = sum(A[i][j]*x[j] for j in range(n) if j != i)
            x_new[i] = (b[i]-s)/A[i][i]
        print('x = ', x_new)
        x_list.append(float(x_new[0]))
        y_list.append(float(x_new[1]))
        z_list.append(float(x_new[2]))
        if np.linalg.norm(x_new - x, ord=np.inf) <= tol:
            break
        x = x_new.copy()
    print(f"{k} iterations required for convergence")
    return x_new

def on_select(event):
    choice = combo.get()

    # Reset all lines to "faded" and thin
    for line in [line_x, line_y, line_z]:
        line.set_alpha(0.2)
        line.set_linewidth(1)

    # Highlight the chosen one
    if choice == "Variable X":
        line_x.set_alpha(1.0)
        line_x.set_linewidth(2)
    elif choice == "Variable Y":
        line_y.set_alpha(1.0)
        line_y.set_linewidth(2)
    elif choice == "Variable Z":
        line_z.set_alpha(1.0)
        line_z.set_linewidth(2)
    else:
        for line in [line_x, line_y, line_z]:
            line.set_alpha(1)
            line.set_linewidth(1)

    canvas.draw()  # Refresh the plot
